14def rp_index_np(data: np.ndarray) -> np.ndarray: 

15 """ 

16 Refers to Section 4.2 RP indexing in the paper. 

17 This function will best match the historical objects with each object in the last frame. 

18 

19 inputs shape: (batch, temporal seq, rp, features) 

20 :param data: 

21 :return: 

22 """ 

23 if data.ndim == 3: 

24 data = np.expand_dims(data, axis=0) 

25 

26 areas = (data[:, :, :, 2] - data[:, :, :, 0]) * (data[:, :, :, 3] - data[:, :, :, 1]) 

27 output = np.zeros(data.shape) 

28 batch_size = data.shape[0] 

29 

30 for batch in range(batch_size): 

31 last_frame = data[batch, -1, :, :] 

32 

33 # evaluate the change of centroid point 

34 x_offset = np.abs(np.subtract.outer( 

35 last_frame[:, 0] + last_frame[:, 2], 

36 data[batch, :, :, 0] + data[batch, :, :, 2] 

37 )) / 2 

38 

39 y_offset = np.abs(np.subtract.outer( 

40 last_frame[:, 1] + last_frame[:, 3], 

41 data[batch, :, :, 1] + data[batch, :, :, 3] 

42 )) / 2 

43 

44 area_diff = np.abs(np.subtract.outer( 

45 np.sqrt(areas[batch, -1, :]), 

46 np.sqrt(areas[batch, :, :]) 

47 )) 

48 

49 metrics = x_offset + y_offset + area_diff # + shape_diff 

50 

51 # Find the smallest metric values 

52 index = np.argmin(metrics, axis=2).swapaxes(0, 1) 

53 

54 output[batch, :, :, :] = np.take_along_axis(data[batch], np.expand_dims(index, axis=-1), axis=1) 

55 

56 # Set a hard threshold for the metric. 

57 metrics_min = np.min(metrics, axis=2) 

58 

59 mask = (metrics_min > threshold).swapaxes(0, 1) 

60 

61 # Reset the RPs running out of the threshold to zero 

62 output[batch, mask] = np.zeros((1, 4)) 

63 

64 # Post-processing: 

65 # Corner case 1: if finding any empty RPs in the last frame, we set the RP in prev frames to empty also. 

66 zero_mask = last_frame[:, 0:4].sum(axis=1) == 0 

67 

68 output[batch, :, zero_mask] = np.zeros((1, 4)) 

69 

70 # Corner case 2: if locating an non-empty RP in the last frame, but an empty position in prev frames, 

71 # Set prev frames to the same RP. 

72 non_zero_mask = (~zero_mask) & (output[batch, :, :, 0:4].sum(axis=2) == 0) 

73 t, index = np.where(non_zero_mask) 

74 output[batch, t, index] = last_frame[index] 

75 

76 return output

111def check_label_matching(data): 

112 ''' 

113 match the rp label along the temporal axis and check the matching accuracy 

114 :param data: data is in the format of [batch, time, rp, feature] 

115 :return: acc (%) of correct label matching 

116 ''' 

117 

118 def _check(test, gt): 

119 non_zero_mask = gt[:, :, 0:4].sum(axis=2) != 0.0 

120 match = gt[non_zero_mask][:, -3] == test[non_zero_mask][:, -3] 

121 return float(sum(match)) / len(match) if len(match) > 0 else 1 

122 

123 last = data[:, -1, :, :] 

124 accs = [] 

125 for i in range(1, data.shape[1]): 

126 test = data[:, i, :, :] 

127 acc = _check(test, last) 

128 accs.append(acc) 

129 return sum(accs) / len(accs)

147def render_bbox(bbox, image, color=(255, 0, 0), thickness=2) -> np.ndarray: 

148 ''' 

149 render output bbox on an image 

150 :param bbox: 

151 :param image: 

152 :param color: 

153 :param thickness: 

154 :return: a rendered result 

155 ''' 

156 output = image.copy() 

157 for box in bbox: 

158 top_left, bottom_right = (0, 0), (0, 0) 

159 if isinstance(box, torch.Tensor): 

160 box = box.to(torch.int64) 

161 top_left, bottom_right = box[:2].tolist(), box[2:].tolist() 

162 elif isinstance(box, np.ndarray): 

163 top_left, bottom_right = box[:2], box[2:] 

164 

165 output = cv2.rectangle( 

166 output, tuple(top_left), tuple(bottom_right), color=tuple(color), thickness=thickness, 

167 ) 

168 return output
